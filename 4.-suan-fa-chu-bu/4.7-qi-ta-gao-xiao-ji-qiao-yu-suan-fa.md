#### 4.7.3 随机选择算法

```cpp
//从一个无序的数组中求出第k大的数（假设数组中的数各不相同），对整个数组排序是做了多余的事

//选取随机主元,对区间[left, right]进行划分
int randPartition(int a[], int left, int right) {
    /*
    先用rand()生成一个[0, RAND_MAX（cstdlib里的一个常数）]范围内的随机整数，
    然后将其映射到[left, right]范围内。
    */
    int r = round(1.0 * rand() / RAND_MAX * (right - left) + left);
    swap(a[left], a[r]);
    
    int pivot = a[left], i = left, j = right;
    while(i < j) {
        while(i < j && a[j] > pivot) j--;
        while(i < j && a[i] <= pivot) i++;
        if(i < j) swap(a[i], a[j]);
    }
    swap(a[i], a[left]);
    return i;
}

//随机选择算法，从a[left, right]中返回第k大的数
int randSelect(int a[], int left, int right, int k) {
    if(left == right) return a[left];
    int p = randPartition(a, left, right);    //划分后主元的位置为p
    int m = p - left + 1;    //主元a[p]是a[left, right]中的第m大
    if(k == m) return a[p];    //主元a[p]正是第k大
    if(k < m) return randSelect(a, left, p - 1, k);    //第k大在主元a[p]左边
    else randSelect(a, p + 1, right, k - m);    //第k大在主元a[p]右边
}
```
